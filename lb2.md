 #### Код программы
```cpp
#include <iostream>
using namespace std;

void tabulate(double a, double b, double x_start, double x_end, double dx)
{
	cout << "x\tz" << endl;
	double max = INT_MIN;
	double min = INT_MAX;
	for (double_t x = x_start; x <= x_end; x = round((x+dx) * 100) / 100)
	{
		double z;
		if (x <= a)
			z = abs(x);
		else if (x > a&& x < b)
			z = abs(x) + cos(x);
		else
			z = tan(x);
		if (z > max)
			max = z;
		if (z < min)
			min = z;
		cout << x << "\t" << z << endl;
	}
	cout << "Max = " << max << endl;
	cout << "Min = " << min << endl;
}
int main()
{
	double a, b, x_start, x_end, dx;
	cout << "a = ";
	cin >> a;
	cout << "b = ";
	cin >> b;
	cout << "x start = ";
	cin >> x_start;
	cout << "x end = ";
	cin >> x_end;
	cout << "dx = ";
	cin >> dx;
	tabulate(a, b, x_start, x_end, dx);
}
```
